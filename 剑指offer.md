    def findDuplicate(self, nums):
        left, right = 1, len(nums)-1
        while left < right:
            mid = (left + right) >> 1
            count = 0
            for i in nums:
                if i <= mid:
                    count += 1
            if count > mid:
                right = mid
            else:
                left = mid + 1
        return left
